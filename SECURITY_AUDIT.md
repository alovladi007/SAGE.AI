    # Validate size
    content = await file.read()
    if len(content) > MAX_FILE_SIZE:
        raise HTTPException(400, "File too large")

    # Validate type
    if file.content_type not in ALLOWED_TYPES:
        raise HTTPException(400, "Invalid file type")
